### user_tb

---

```sql

create table application_tb
(
    company_id integer,
    id         integer generated by default as identity,
    job_id     integer,
    resume_id  integer,
    user_id    integer,
    created_at timestamp(6),
    status     varchar(255),
    primary key (id),
    constraint uk_user_company unique (user_id, company_id)
)

 ```
